# QW Config Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a TypeScript library for QuakeWorld config parsing, documentation, and cross-client conversion, plus a "My Quake" tab in slipgate-app with config viewer, comparison, and converter UI.

**Architecture:** `packages/qw-config/` is a standalone TypeScript library (data + logic) consumed by `apps/slipgate-app/` via Bun workspace. The library handles cvar databases, config parsing, agnostic format, and conversion. The app provides the UI: config viewer with categorized cvar list, side-by-side comparison, and converter with gap report.

**Tech Stack:** TypeScript, SolidJS, Tailwind 4, DaisyUI 5, Tauri 2, Bun workspaces, lucide-solid icons

```typescript
import type { CvarInfo, CategoryGroup } from "../types.js";
import variablesJson from "../data/ezquake-variables.json";

interface EzGroup {
  id: string;
  "major-group": string;
  name: string;
}

interface EzVariable {
  type: string;
  "group-id": string;
  desc?: string;
  default?: string;
  remarks?: string;
  values?: { name: string; description: string }[];
}

interface EzVariablesFile {
  groups: EzGroup[];
  vars: Record<string, EzVariable>;
}

const data = variablesJson as unknown as EzVariablesFile;

/** Build a group-id → { category, group } lookup from ezQuake's group hierarchy */
function buildGroupLookup(): Map<string, { category: string; group: string }> {
  const lookup = new Map<string, { category: string; group: string }>();
  for (const g of data.groups) {
    lookup.set(g.id, {
      category: g["major-group"],
      group: g.name,
    });
  }
  return lookup;
}

/** Extract the category hierarchy (major groups → sub-groups) */
export function getEzQuakeCategories(): CategoryGroup[] {
  const majorGroups = new Map<string, Set<string>>();
  for (const g of data.groups) {
    if (!majorGroups.has(g["major-group"])) {
      majorGroups.set(g["major-group"], new Set());
    }
    majorGroups.get(g["major-group"])!.add(g.name);
  }
  return Array.from(majorGroups.entries()).map(([name, groups]) => ({
    name,
    groups: Array.from(groups),
  }));
}

/** Load all ezQuake cvars as CvarInfo entries */
export function loadEzQuakeCvars(): Map<string, CvarInfo> {
  const groupLookup = buildGroupLookup();
  const cvars = new Map<string, CvarInfo>();

  for (const [name, v] of Object.entries(data.vars)) {
    const groupInfo = groupLookup.get(v["group-id"]);
    const cvar: CvarInfo = {
      name,
      description: v.desc ?? "",
      remarks: v.remarks,
      type: v.type as CvarInfo["type"],
      default: v.default,
      values: v.values?.map((val) => ({
        name: val.name,
        description: val.description,
      })),
      category: groupInfo?.category ?? "Other",
      group: groupInfo?.group ?? "Other",
      client: "ezquake",
    };
    cvars.set(name, cvar);
  }

  return cvars;
}

/** Get the total count of ezQuake variables */
export function getEzQuakeVarCount(): number {
  return Object.keys(data.vars).length;
}
```

- [ ] **Step 1: Write FTE cvar extraction script**

Create `packages/qw-config/scripts/extract-fte-cvars.ts`:

```typescript
/**
 * Extracts cvar definitions from FTE engine C source files.
 * Parses CVARD, CVARFD, CVARAFD macros to get name, default, description.
 *
 * Usage: bun run scripts/extract-fte-cvars.ts
 */
import { readdir, readFile, writeFile } from "fs/promises";
import { join } from "path";

const FTE_ENGINE_DIR = join(import.meta.dir, "../../../research/repos/fteqw/engine");
const OUTPUT_FILE = join(import.meta.dir, "../src/data/fte-variables.json");

interface FteCvar {
  name: string;
  default: string;
  description: string;
  alias?: string;
  flags?: string;
  sourceFile: string;
  group?: string;
}

/** Recursively find all .c and .h files */
async function findCFiles(dir: string): Promise<string[]> {
  const files: string[] = [];
  const entries = await readdir(dir, { withFileTypes: true });
  for (const entry of entries) {
    const full = join(dir, entry.name);
    if (entry.isDirectory()) {
      files.push(...(await findCFiles(full)));
    } else if (entry.name.endsWith(".c") || entry.name.endsWith(".h")) {
      files.push(full);
    }
  }
  return files;
}

/**
 * Extract cvar group registrations: Cvar_Register(&cvar, group_string)
 * and cvargroup_* definitions
 */
function extractGroups(content: string): Map<string, string> {
  const groups = new Map<string, string>();

  // Match: char cvargroup_foo[] = "description";
  // Match: #define cvargroup_foo "description"
  const groupDefs = new Map<string, string>();
  for (const m of content.matchAll(/(?:char\s+|#define\s+)(cvargroup_\w+)(?:\[\])?\s*=?\s*"([^"]+)"/g)) {
    groupDefs.set(m[1], m[2]);
  }

  // Match: Cvar_Register(&var_name, cvargroup_name);
  for (const m of content.matchAll(/Cvar_Register\s*\(\s*&(\w+)\s*,\s*(\w+)/g)) {
    const varName = m[1];
    const groupRef = m[2];
    const groupDesc = groupDefs.get(groupRef) ?? groupRef;
    groups.set(varName, groupDesc);
  }

  return groups;
}

/** Extract cvars from a single C file */
function extractFromFile(content: string, relativePath: string): FteCvar[] {
  const cvars: FteCvar[] = [];
  const groups = extractGroups(content);

  // CVARD("name", "default", "description")
  // Description may span multiple lines with string concatenation
  const cvardPattern = /(\w+)\s*=\s*CVARD\s*\(\s*"([^"]+)"\s*,\s*"([^"]*)"\s*,\s*"((?:[^"\\]|\\.|"\s*")*?)"\s*\)/gs;
  for (const m of content.matchAll(cvardPattern)) {
    const varIdent = m[1];
    cvars.push({
      name: m[2],
      default: m[3],
      description: m[4].replace(/"\s*"/g, "").replace(/\\n/g, "\n").replace(/\\"/g, '"'),
      sourceFile: relativePath,
      group: groups.get(varIdent),
    });
  }

  // CVARFD("name", "default", FLAGS, "description")
  const cvarfdPattern = /(\w+)\s*=\s*CVARFD\s*\(\s*"([^"]+)"\s*,\s*"([^"]*)"\s*,\s*([^,]+?)\s*,\s*"((?:[^"\\]|\\.|"\s*")*?)"\s*\)/gs;
  for (const m of content.matchAll(cvarfdPattern)) {
    const varIdent = m[1];
    cvars.push({
      name: m[2],
      default: m[3],
      description: m[5].replace(/"\s*"/g, "").replace(/\\n/g, "\n").replace(/\\"/g, '"'),
      flags: m[4].trim(),
      sourceFile: relativePath,
      group: groups.get(varIdent),
    });
  }

  // CVARAFD("name", "default", "alias", FLAGS, "description")
  const cvarafdPattern = /(\w+)\s*=\s*CVARAFD\s*\(\s*"([^"]+)"\s*,\s*"([^"]*)"\s*,\s*(?:\/\*[^*]*\*\/\s*)?"([^"]+)"\s*,\s*([^,]+?)\s*,\s*"((?:[^"\\]|\\.|"\s*")*?)"\s*\)/gs;
  for (const m of content.matchAll(cvarafdPattern)) {
    const varIdent = m[1];
    cvars.push({
      name: m[2],
      default: m[3],
      alias: m[4],
      description: m[6].replace(/"\s*"/g, "").replace(/\\n/g, "\n").replace(/\\"/g, '"'),
      flags: m[5].trim(),
      sourceFile: relativePath,
      group: groups.get(varIdent),
    });
  }

  return cvars;
}

async function main() {
  console.log(`Scanning FTE source: ${FTE_ENGINE_DIR}`);
  const files = await findCFiles(FTE_ENGINE_DIR);
  console.log(`Found ${files.length} C/H files`);

  const allCvars: FteCvar[] = [];
  for (const file of files) {
    const content = await readFile(file, "utf-8");
    const relativePath = file.replace(FTE_ENGINE_DIR + "/", "");
    const cvars = extractFromFile(content, relativePath);
    allCvars.push(...cvars);
  }

  // Deduplicate by name (keep first occurrence)
  const seen = new Set<string>();
  const unique = allCvars.filter((c) => {
    if (seen.has(c.name)) return false;
    seen.add(c.name);
    return true;
  });

  console.log(`Extracted ${unique.length} unique cvars (${allCvars.length} total including dupes)`);

  await writeFile(OUTPUT_FILE, JSON.stringify(unique, null, 2));
  console.log(`Written to ${OUTPUT_FILE}`);
}

main().catch(console.error);
```

```typescript
import type { CvarInfo } from "../types.js";
import fteData from "../data/fte-variables.json";

interface FteRawCvar {
  name: string;
  default: string;
  description: string;
  alias?: string;
  flags?: string;
  sourceFile: string;
  group?: string;
}

const data = fteData as FteRawCvar[];

/**
 * Map FTE group strings to ezQuake-compatible major categories.
 * FTE uses descriptive strings like "server physics variables".
 * We map these to our standard categories.
 */
function mapFteGroup(group?: string): { category: string; group: string } {
  if (!group) return { category: "Other", group: "Uncategorized" };

  const g = group.toLowerCase();
  if (g.includes("physics")) return { category: "Server", group: "Physics" };
  if (g.includes("permission")) return { category: "Server", group: "Permissions" };
  if (g.includes("serverinfo")) return { category: "Server", group: "Server Info" };
  if (g.includes("control")) return { category: "Server", group: "Control" };
  if (g.includes("ranking")) return { category: "Server", group: "Rankings" };
  // Most client cvars won't have group info — categorize by source file
  return { category: "Other", group: group };
}

/**
 * Infer category from FTE source file path when no group is available.
 */
function inferCategory(sourceFile: string): { category: string; group: string } {
  if (sourceFile.includes("cl_")) return { category: "Client", group: "Client Settings" };
  if (sourceFile.includes("gl_") || sourceFile.includes("/gl/")) return { category: "Graphics", group: "Rendering" };
  if (sourceFile.includes("sbar")) return { category: "HUD", group: "Scoreboard" };
  if (sourceFile.includes("view")) return { category: "Graphics", group: "View" };
  if (sourceFile.includes("in_")) return { category: "Input", group: "Input" };
  if (sourceFile.includes("snd") || sourceFile.includes("s_")) return { category: "Sound", group: "Sound" };
  if (sourceFile.includes("sv_")) return { category: "Server", group: "Server" };
  if (sourceFile.includes("net_")) return { category: "Network", group: "Network" };
  if (sourceFile.includes("pr_")) return { category: "Server", group: "Progs" };
  return { category: "Other", group: "Other" };
}

/** Load all FTE cvars as CvarInfo entries */
export function loadFteCvars(): Map<string, CvarInfo> {
  const cvars = new Map<string, CvarInfo>();

  for (const raw of data) {
    const groupInfo = raw.group ? mapFteGroup(raw.group) : inferCategory(raw.sourceFile);
    const cvar: CvarInfo = {
      name: raw.name,
      description: raw.description,
      type: inferType(raw.default, raw.description),
      default: raw.default,
      category: groupInfo.category,
      group: groupInfo.group,
      client: "fte",
    };
    cvars.set(raw.name, cvar);
  }

  return cvars;
}

/** Infer type from default value and description */
function inferType(defaultVal: string, desc: string): CvarInfo["type"] {
  if (defaultVal === "0" || defaultVal === "1") {
    // Check if description suggests boolean
    const d = desc.toLowerCase();
    if (d.includes("enable") || d.includes("disable") || d.includes("if set") || d.includes("toggle")) {
      return "boolean";
    }
  }
  if (defaultVal === "" || /[a-zA-Z]/.test(defaultVal) && !/^\d/.test(defaultVal)) {
    return "string";
  }
  if (defaultVal.includes(".")) return "float";
  if (/^\d+$/.test(defaultVal)) return "integer";
  return "string";
}

/** Get the total count of FTE cvars */
export function getFteCvarCount(): number {
  return data.length;
}
```

```typescript
/**
 * Assembles the QWCL baseline cvar list.
 * QWCL cvars are the ~189 original id Software cvars that all clients inherit.
 * We identify them by cross-referencing the original QWCL source with ezQuake's documentation.
 *

import { join } from "path";

const QWCL_SOURCE = join(import.meta.dir, "../../../research/repos/qwcl-original/QW/client");
const EZQUAKE_VARS = join(import.meta.dir, "../src/data/ezquake-variables.json");
const OUTPUT = join(import.meta.dir, "../src/data/qwcl-variables.json");

interface QwclCvar {
  name: string;
  default: string;
  description: string;
  category: string;
  /** Source of the description */
  descriptionSource: "ezquake" | "fte" | "manual" | "self-explanatory";
}

async function main() {
  // Step 1: Extract cvar names from QWCL source (Cvar_RegisterVariable calls)
  const qwclNames = new Set<string>();

  let files: string[];
  try {
    files = (await readdir(QWCL_SOURCE)).filter((f) => f.endsWith(".c") || f.endsWith(".h"));
  } catch {
    console.error(`QWCL source not found at ${QWCL_SOURCE}`);
    console.error("Run: cd research && ./update-repos.sh");
    process.exit(1);
  }

  for (const file of files) {
    const content = await readFile(join(QWCL_SOURCE, file), "utf-8");
    // Match Cvar_RegisterVariable patterns and cvar_t declarations
    for (const m of content.matchAll(/["'](\w+)["']\s*,\s*["']([^"']*)["']/g)) {
      // Heuristic: looks like a cvar registration if first arg is lowercase
      if (m[1] === m[1].toLowerCase() && m[1].length > 1) {
        qwclNames.add(m[1]);
      }
    }
    // Also match: {"name", "default"} style initializers
    for (const m of content.matchAll(/\{\s*"(\w+)"\s*,\s*"([^"]*)"/g)) {
      if (m[1] === m[1].toLowerCase()) {
        qwclNames.add(m[1]);
      }
    }
  }

  // Step 2: Cross-reference with ezQuake for descriptions
  const ezData = JSON.parse(await readFile(EZQUAKE_VARS, "utf-8"));
  const ezVars = ezData.vars as Record<string, { desc?: string; default?: string; "group-id": string }>;

  const cvars: QwclCvar[] = [];
  for (const name of qwclNames) {
    const ez = ezVars[name];
    cvars.push({
      name,
      default: ez?.default ?? "",
      description: ez?.desc ?? "",
      category: "QWCL",
      descriptionSource: ez?.desc ? "ezquake" : "manual",
    });
  }

  cvars.sort((a, b) => a.name.localeCompare(b.name));

  console.log(`Assembled ${cvars.length} QWCL cvars (${cvars.filter((c) => c.description).length} with descriptions)`);
  await writeFile(OUTPUT, JSON.stringify(cvars, null, 2));
  console.log(`Written to ${OUTPUT}`);
}

main().catch(console.error);
```

```typescript
import type { ClientId, CvarInfo, CvarDatabase, CvarMapping, CategoryGroup } from "../types.js";
import { loadEzQuakeCvars, getEzQuakeCategories } from "./ezquake.js";
import { loadFteCvars } from "./fte.js";
import { loadQwclCvars } from "./qwcl.js";

let cachedDb: CvarDatabase | null = null;

/** Load the complete cvar database (all clients + mappings). Cached after first call. */
export function loadDatabase(): CvarDatabase {
  if (cachedDb) return cachedDb;

  const ezCvars = loadEzQuakeCvars();
  const fteCvars = loadFteCvars();
  const qwclCvars = loadQwclCvars();

  // Auto-generate mappings for cvars with the same name across clients
  const mappings = generateAutoMappings(ezCvars, fteCvars, qwclCvars);

  // Use ezQuake categories as the baseline taxonomy
  const categories = getEzQuakeCategories();

  cachedDb = {
    clients: {
      ezquake: ezCvars,
      fte: fteCvars,
      qwcl: qwclCvars,
    },
    mappings,
    categories,
  };

  return cachedDb;
}

/**
 * Auto-generate mappings for cvars that share the same name across clients.
 * These are the "shared baseline" — QWCL heritage cvars like sensitivity, fov, etc.
 */
function generateAutoMappings(
  ezCvars: Map<string, CvarInfo>,
  fteCvars: Map<string, CvarInfo>,
  qwclCvars: Map<string, CvarInfo>,
): CvarMapping[] {
  const allNames = new Set<string>();
  for (const name of ezCvars.keys()) allNames.add(name);
  for (const name of fteCvars.keys()) allNames.add(name);
  for (const name of qwclCvars.keys()) allNames.add(name);

  const mappings: CvarMapping[] = [];
  for (const name of allNames) {
    const clients: Partial<Record<ClientId, string>> = {};
    if (ezCvars.has(name)) clients.ezquake = name;
    if (fteCvars.has(name)) clients.fte = name;
    if (qwclCvars.has(name)) clients.qwcl = name;

    // Only create a mapping if the cvar exists in 2+ clients
    const clientCount = Object.keys(clients).length;
    if (clientCount >= 2) {
      mappings.push({ id: name, clients });
    }
  }

  return mappings;
}

/** Look up a cvar's info, preferring ezQuake's description (richest source) */
export function lookupCvar(name: string, client?: ClientId): CvarInfo | undefined {
  const db = loadDatabase();
  if (client) {
    return db.clients[client].get(name);
  }
  // Try ezQuake first (best descriptions), then FTE, then QWCL
  return db.clients.ezquake.get(name) ?? db.clients.fte.get(name) ?? db.clients.qwcl.get(name);
}

/** Find the equivalent cvar name in another client */
export function findEquivalent(cvarName: string, sourceClient: ClientId, targetClient: ClientId): string | undefined {
  const db = loadDatabase();
  const mapping = db.mappings.find((m) => m.clients[sourceClient] === cvarName);
  return mapping?.clients[targetClient];
}
```

```typescript
import type { ParsedConfig } from "../types.js";

/**
 * Parse a QuakeWorld config file (config.cfg / fte.cfg / autoexec.cfg).
 * Handles: cvar value, bind key action, alias name commands, exec filename.
 * All QW clients share the same config file syntax.
 */
export function parseConfig(text: string): ParsedConfig {
  const cvars = new Map<string, string>();
  const binds = new Map<string, string>();
  const aliases = new Map<string, string>();
  const execs: string[] = [];
  const unparsed: string[] = [];

  for (const rawLine of text.split("\n")) {
    const line = rawLine.trim();

    // Skip empty lines and comments
    if (!line || line.startsWith("//")) continue;

    // Try to parse as a known command
    const parsed =
      parseBind(line, binds) ||
      parseAlias(line, aliases) ||
      parseExec(line, execs) ||
      parseCvar(line, cvars);

    if (!parsed) {
      unparsed.push(line);
    }
  }

  return { cvars, binds, aliases, execs, unparsed };
}

/** Parse: bind <key> "<action>" */
function parseBind(line: string, binds: Map<string, string>): boolean {
  const match = line.match(/^bind\s+(\S+)\s+"?(.*?)"?\s*$/i);
  if (!match) return false;
  const key = match[1].toLowerCase();
  const action = match[2];
  binds.set(key, action);
  return true;
}

/** Parse: alias <name> "<commands>" */
function parseAlias(line: string, aliases: Map<string, string>): boolean {
  const match = line.match(/^alias\s+(\S+)\s+"(.*?)"\s*$/i);
  if (!match) return false;
  aliases.set(match[1], match[2]);
  return true;
}

/** Parse: exec <filename> */
function parseExec(line: string, execs: string[]): boolean {
  const match = line.match(/^exec\s+(\S+)\s*$/i);
  if (!match) return false;
  execs.push(match[1]);
  return true;
}

/** Parse: <cvar> "<value>" or <cvar> <value> */
function parseCvar(line: string, cvars: Map<string, string>): boolean {
  // Quoted value: cvar "value with spaces"
  const quotedMatch = line.match(/^(\w+)\s+"(.*?)"\s*$/);
  if (quotedMatch) {
    cvars.set(quotedMatch[1], quotedMatch[2]);
    return true;
  }

  // Unquoted value: cvar value
  const unquotedMatch = line.match(/^(\w+)\s+(\S+)\s*$/);
  if (unquotedMatch) {
    cvars.set(unquotedMatch[1], unquotedMatch[2]);
    return true;
  }

  return false;
}
```

```typescript
import type { ClientId, ParsedConfig, CvarMapping } from "../types.js";
import { loadDatabase } from "../loaders/index.js";

export interface ConversionResult {
  /** Cvars successfully converted to target client */
  targetCvars: Map<string, string>;
  /** Binds carried over */
  targetBinds: Map<string, string>;
  /** Source cvars that transferred (same name in both clients) */
  transferred: Array<{ cvar: string; value: string }>;
  /** Source cvars mapped to different target name */
  mapped: Array<{ sourceCvar: string; targetCvar: string; value: string; note?: string }>;
  /** Source cvars with no equivalent in target */
  noEquivalent: Array<{ cvar: string; value: string }>;
}

/**
 * Convert a parsed config from one client to another.
 * Uses the cvar database and mappings to translate cvar names.
 * Binds are carried over as-is (same syntax across clients).
 */
export function convertConfig(
  parsed: ParsedConfig,
  sourceClient: ClientId,
  targetClient: ClientId,
): ConversionResult {
  const db = loadDatabase();
  const targetCvars = new Map<string, string>();
  const transferred: ConversionResult["transferred"] = [];
  const mapped: ConversionResult["mapped"] = [];
  const noEquivalent: ConversionResult["noEquivalent"] = [];

  for (const [cvarName, value] of parsed.cvars) {
    // Check if the cvar exists in the target client with the same name
    if (db.clients[targetClient].has(cvarName)) {
      targetCvars.set(cvarName, value);
      transferred.push({ cvar: cvarName, value });
      continue;
    }

    // Check manual + auto mappings for a different-name equivalent
    const mapping = db.mappings.find(
      (m: CvarMapping) => m.clients[sourceClient] === cvarName && m.clients[targetClient],
    );
    if (mapping) {
      const targetName = mapping.clients[targetClient]!;
      targetCvars.set(targetName, value);
      mapped.push({
        sourceCvar: cvarName,
        targetCvar: targetName,
        value,
        note: mapping.note,
      });
      continue;
    }

    // No equivalent found
    noEquivalent.push({ cvar: cvarName, value });
  }

  // Binds carry over as-is — same syntax across all QW clients
  const targetBinds = new Map(parsed.binds);

  return { targetCvars, targetBinds, transferred, mapped, noEquivalent };
}
```

```typescript
/**
```

```typescript
 * FTE uses the same syntax as ezQuake/QWCL: cvar "value" and bind key "action".

export function writeFteConfig(result: ConversionResult): string {

  lines.push("// Generated by Slipgate QW Config Converter");

  lines.push("");
  const sortedCvars = Array.from(result.targetCvars.entries()).sort((a, b) => a[0].localeCompare(b[0]));
  for (const [name, value] of sortedCvars) {
    if (value.includes(" ")) {
      lines.push(`${name} "${value}"`);
    } else {
      lines.push(`${name} "${value}"`);
    }
  }

  // Write binds
  if (result.targetBinds.size > 0) {
    lines.push("");
    lines.push("// ── Binds ──");
    lines.push("");
    const sortedBinds = Array.from(result.targetBinds.entries()).sort((a, b) => a[0].localeCompare(b[0]));
    for (const [key, action] of sortedBinds) {
      lines.push(`bind ${key} "${action}"`);
    }
  }

  lines.push("");
  return lines.join("\n");
```

```typescript
  onBack: () => void;

export default function ConfigCompare(props: ConfigCompareProps) {
  const [filter, setFilter] = createSignal<DiffFilter>("all");
  const [searchQuery, setSearchQuery] = createSignal("");
  const [hideDefaults, setHideDefaults] = createSignal(false);

  const rightParsed = createMemo(() => parseConfig(props.rightConfigText));

  const allRows = createMemo(() => {
    const left = props.leftConfig;
    const right = rightParsed().cvars;
    const allNames = new Set([...Object.keys(left), ...right.keys()]);
    const rows: CompareRow[] = [];

    for (const name of allNames) {
      const lv = left[name] ?? null;
      const rv = right.get(name) ?? null;
      const info = lookupCvar(name);
      const isDefault = info?.default !== undefined && lv === info.default && rv === info.default;

      rows.push({
        name,
        leftValue: lv,
        rightValue: rv,
        description: info?.description ?? "",
        isDifferent: lv !== rv,
      });
    }

    return rows.sort((a, b) => a.name.localeCompare(b.name));
  });

  const diffCount = createMemo(() => allRows().filter((r) => r.isDifferent).length);

  const filteredRows = createMemo(() => {
    let rows = allRows();

    switch (filter()) {
      case "differences":
        rows = rows.filter((r) => r.isDifferent);
        break;
      case "same":
        rows = rows.filter((r) => !r.isDifferent);
        break;
      case "only_left":
        rows = rows.filter((r) => r.leftValue !== null && r.rightValue === null);
        break;
      case "only_right":
        rows = rows.filter((r) => r.leftValue === null && r.rightValue !== null);
        break;
    }

    if (hideDefaults()) {
      rows = rows.filter((r) => {
        const info = lookupCvar(r.name);
        return !(info?.default !== undefined && r.leftValue === info.default && r.rightValue === info.default);
      });
    }

    const q = searchQuery().toLowerCase();
    if (q) {
      rows = rows.filter((r) => r.name.toLowerCase().includes(q) || r.description.toLowerCase().includes(q));
    }

    return rows;
  });

  return (
    <div class="flex flex-col h-full">
      {/* Header */}
      <div class="flex border-b border-base-300 bg-base-200">
        <div class="flex-1 flex items-center gap-2 px-3 py-2 border-r border-base-300">
          <button class="btn btn-xs btn-ghost" onClick={props.onBack}>
            <ArrowLeft size={14} />
          </button>
          <span class="font-mono text-sm text-warning font-semibold">{props.leftLabel}</span>
        </div>
        <div class="flex-1 flex items-center gap-2 px-3 py-2">
          <span class="font-mono text-sm text-secondary font-semibold">{props.rightLabel}</span>
        </div>
      </div>

      {/* Filter bar */}
      <div class="flex items-center gap-2 px-3 py-1.5 border-b border-base-300/50">
        <div class="flex gap-1">
          {(["all", "differences", "same", "only_left", "only_right"] as DiffFilter[]).map((f) => (
            <button
              class={`badge badge-sm cursor-pointer ${filter() === f ? (f === "differences" ? "badge-error" : "badge-warning") : "badge-ghost"}`}
              onClick={() => setFilter(f)}
            >
              {f === "all" && `All (${allRows().length})`}
              {f === "differences" && `Differences (${diffCount()})`}
              {f === "same" && "Same"}
              {f === "only_left" && "Only left"}
              {f === "only_right" && "Only right"}
            </button>
          ))}
        </div>
        <div class="ml-auto flex items-center gap-2">
          <label class="flex items-center gap-1 text-xs text-base-content/50 cursor-pointer">
            <input type="checkbox" class="checkbox checkbox-xs" checked={hideDefaults()} onChange={(e) => setHideDefaults(e.currentTarget.checked)} />
            Hide defaults
          </label>
          <input type="text" placeholder="Search..." class="input input-xs input-bordered w-28" value={searchQuery()} onInput={(e) => setSearchQuery(e.currentTarget.value)} />
        </div>
      </div>

      {/* Comparison rows */}
      <div class="flex-1 overflow-y-auto">
        <For each={filteredRows()}>
          {(row) => (
            <div class={`flex border-b border-base-300/30 ${row.isDifferent ? "bg-error/5" : ""}`}>
              <div class="flex-1 flex items-center px-3 py-1.5 border-r border-base-300/30">
                <span class="w-32 font-mono text-xs" classList={{ "text-warning": row.leftValue !== null, "text-base-content/20": row.leftValue === null }}>
                  {row.name}
                </span>
                <span class="w-14 font-mono text-xs text-base-content/70" classList={{ "text-error font-semibold": row.isDifferent }}>
                  {row.leftValue ?? ""}
                </span>
                <span class="flex-1 text-xs text-base-content/40 truncate">{row.description}</span>
              </div>
              <div class="flex-1 flex items-center px-3 py-1.5">
                <span class="w-32 font-mono text-xs" classList={{ "text-secondary": row.rightValue !== null, "text-base-content/20": row.rightValue === null }}>
                  {row.name}
                </span>
                <span class="w-14 font-mono text-xs text-base-content/70" classList={{ "text-error font-semibold": row.isDifferent }}>
                  {row.rightValue ?? ""}
                </span>
                {row.leftValue === null && <span class="text-xs text-base-content/20 italic">not in left</span>}
                {row.rightValue === null && row.leftValue !== null && ""}
              </div>
            </div>
          )}
        </For>
      </div>
    </div>
  );
}
```

```typescript
import { ArrowLeft, Download } from "lucide-solid";
import { parseConfig, convertConfig, generateReport, writeFteConfig } from "qw-config";
import type { ConversionReport, ConvertedCvar } from "qw-config";

type ReportFilter = "all" | "transferred" | "mapped" | "no_equivalent";

interface ConfigConverterProps {
  rawCvars: Record<string, string>;
  configName: string;
  onBack: () => void;
}

export default function ConfigConverter(props: ConfigConverterProps) {
  const [filter, setFilter] = createSignal<ReportFilter>("all");
  const [searchQuery, setSearchQuery] = createSignal("");

  // Build a minimal config text from raw cvars for parsing
  const configText = createMemo(() =>
    Object.entries(props.rawCvars)
      .map(([k, v]) => `${k} "${v}"`)
      .join("\n"),
  );

  const parsed = createMemo(() => parseConfig(configText()));
  const result = createMemo(() => convertConfig(parsed(), "ezquake", "fte"));
  const report = createMemo(() => generateReport(result(), parsed(), "ezquake", "fte"));

  const allItems = createMemo((): ConvertedCvar[] => {
    const r = report();
    switch (filter()) {
      case "transferred": return r.transferred;
      case "mapped": return r.mapped;
      case "no_equivalent": return r.noEquivalent;
      default: return [...r.transferred, ...r.mapped, ...r.noEquivalent];
    }
  });

  const filteredItems = createMemo(() => {
    const q = searchQuery().toLowerCase();
    if (!q) return allItems();
    return allItems().filter(
      (i) => i.sourceCvar.toLowerCase().includes(q) || i.description.toLowerCase().includes(q),
    );
  });

  async function exportFteCfg() {
    const text = writeFteConfig(result());
    // Use Tauri save dialog
    const { save } = await import("@tauri-apps/plugin-dialog");
    const path = await save({ defaultPath: "fte.cfg", filters: [{ name: "Config", extensions: ["cfg"] }] });
    if (path) {
      const { writeTextFile } = await import("@tauri-apps/plugin-fs");
      await writeTextFile(path, text);
    }
  }

    const lines = [
      "# FTE Gap Report",

      `## Mapped to Different Name (${r.mapped.length} cvars)`,
      "",
      ...r.mapped.map((c) => `- **${c.sourceCvar}** → **${c.targetCvar}** = \`${c.sourceValue}\` — ${c.note ?? c.description}`),

    const text = lines.join("\n");

    const { save } = await import("@tauri-apps/plugin-dialog");
    const path = await save({ defaultPath: "fte-gap-report.md", filters: [{ name: "Markdown", extensions: ["md"] }] });

      const { writeTextFile } = await import("@tauri-apps/plugin-fs");
      await writeTextFile(path, text);

        <span class="text-base-content/40">ezQuake</span>

          <div class="text-xs text-base-content/40 uppercase">Transferred</div>
        </div>
```
